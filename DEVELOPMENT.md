### 5. 表格化输出 (v3.0.7 新增)

```python
def render_table(headers: List[str], rows: List[List[str]]) -> str:
    """
    渲染 ASCII 表格
    
    Args:
        headers: 表头列表
        rows: 数据行列表
    
    Returns:
        格式化的表格字符串
    """
    # 计算每列最大宽度
    col_widths = [len(h) for h in headers]
    for row in rows:
        for i, cell in enumerate(row):
            col_widths[i] = max(col_widths[i], len(cell))
    
    # 构建表格
    lines = []
    # 顶线
    lines.append('┌' + '┬'.join('─' * (w + 2) for w in col_widths) + '┐')
    # 表头
    lines.append('│' + '│'.join(f' {h.center(w)} ' for h, w in zip(headers, col_widths)) + '│')
    # 分隔线
    lines.append('├' + '┼'.join('─' * (w + 2) for w in col_widths) + '┤')
    # 数据行
    for row in rows:
        lines.append('│' + '│'.join(f' {c.ljust(w)} ' for c, w in zip(row, col_widths)) + '│')
    # 底线
    lines.append('└' + '┴'.join('─' * (w + 2) for w in col_widths) + '┘')
    
    return '\n'.join(lines)
```
